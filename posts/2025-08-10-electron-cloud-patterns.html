🔧 Simple EM Pattern Sketcher

import numpy as np
import matplotlib.pyplot as plt

def s_orbital(x, y):
    return np.exp(-(x**2 + y**2))

def p_orbital(x, y):
    return x * np.exp(-(x**2 + y**2))

grid = np.linspace(-3, 3, 300)
X, Y = np.meshgrid(grid, grid)
Z = p_orbital(X, Y)

plt.imshow(Z, extent=(-3,3,-3,3), cmap='inferno')
plt.title("P-Orbital EM Pattern")
plt.colorbar()
plt.show()
